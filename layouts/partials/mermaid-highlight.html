{{- /* Lightweight Mermaid syntax highlighter (Chroma has no mermaid lexer). Input: the raw diagram source string. Output: a .chroma 
 whose spans
       reuse the Catppuccin token classes from chroma.css.
       Keyword pass runs FIRST, before any class="..." spans are injected, so the
       keyword "class" cannot match inside the markup we add. */ -}}
{{- $h := . | htmlEscape -}}
{{- $h = $h | replaceRE `\b(flowchart|graph|subgraph|end|direction|sequenceDiagram|participant|actor|loop|alt|opt|else|par|note|over|activate|deactivate|classDiagram|classDef|class|stateDiagram-v2|stateDiagram|state|erDiagram|gantt|pie|journey|gitGraph|TB|TD|BT|RL|LR)\b` `$1` -}}
{{- $h = $h | replaceRE `(?m)(%%.*$)` `$1` -}}
{{- $h = $h | replaceRE `("[^&]*?")` `$1` -}}
{{- $h = $h | replaceRE `(--?(?:>){1,2}|-\.-(?:>)|={2,3}(?:>)|--[xo]|:::|\|)` `$1` -}}

{{ $h | safeHTML }}
